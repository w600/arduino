## Combine gc-sections, archives, and objects
#$(ELF) $(ELFFLAGS) $(LDFLAGS) -o $(GOAL).elf -Wl,-Map,"$(GOAL).map" $(OBJ_FILES) -Wl,--start-group $(AR_TAGETS) -Wl,--end-group
